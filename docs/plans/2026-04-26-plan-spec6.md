use sha2::{Sha256, Digest};

/// Compute SHA-256 of a file. Returns lowercase hex string.
pub fn sha256_file(path: &Path) -> anyhow::Result<String> {
    let bytes = std::fs::read(path)?;
    let hash = Sha256::digest(&bytes);
    Ok(format!("{:x}", hash))
}

// redist/crates/redist-report/src/rplan.rs

use std::collections::HashMap;
use serde::{Serialize, Deserialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct PlanManifest {
    pub rplan_version: String,          // at root of RPLAN
    pub state: String,
    pub year: String,
    pub label: String,
    pub chamber: Option<String>,
    pub num_districts: usize,
    pub balance_tolerance_pct: f64,
    pub seed: Option<u64>,
    pub created_at: String,             // ISO 8601
    pub tiger_sha256: Option<String>,   // SHA-256 of source shapefile
    pub adj_bin_sha256: Option<String>, // SHA-256 of .adj.bin
    pub binary_version: String,
    pub binary_build_commit: Option<String>,
    pub binary_download_url: Option<String>,    // GitHub Releases page URL
    pub binary_release_sha256: Option<String>,  // SHA-256 from GitHub Releases
    pub external_analyzers: Vec<crate::audit::ExternalAnalyzerRecord>,
    pub source: Option<String>,                 // "generated" | "imported"
    pub source_file: Option<String>,            // set when source="imported"
}

#[derive(Debug, Serialize, Deserialize)]
pub struct RplanDocument {
    pub rplan_version: String,
    pub metadata: PlanManifest,
    pub assignments: HashMap<String, usize>,
    pub geometry: Option<serde_json::Value>,    // Optional GeoJSON FeatureCollection
}

/// Validate all assignment GEOIDs: must be exactly 11 ASCII digits.
pub fn validate_rplan_assignments(
    assignments: &HashMap<String, usize>,
) -> anyhow::Result<()> {
    for geoid in assignments.keys() {
        if geoid.len() != 11 {
            anyhow::bail!("GEOID {geoid:?} must be 11 chars (found {})", geoid.len());
        }
        if !geoid.chars().all(|c| c.is_ascii_digit()) {
            anyhow::bail!("GEOID {geoid:?} must be all digits");
        }
    }
    Ok(())
}

/// Validate that the RPLAN file's rplan_version appears at root and not in metadata JSON.
pub fn validate_rplan_version_placement(raw_json: &str) -> anyhow::Result<()> {
    let v: serde_json::Value = serde_json::from_str(raw_json)?;
    if !v["rplan_version"].is_string() {
        anyhow::bail!("rplan_version missing from root of RPLAN document");
    }
    if !v["metadata"]["rplan_version"].is_null() {
        anyhow::bail!("rplan_version must appear only at root, not inside metadata");
    }
    Ok(())
}

/// Validate 100% tract coverage: assignments must contain every expected GEOID.
pub fn validate_coverage(
    assignments: &HashMap<String, usize>,
    expected_geoids: &[String],
) -> anyhow::Result<()> {
    let missing: Vec<_> = expected_geoids.iter()
        .filter(|g| !assignments.contains_key(*g))
        .collect();
    if !missing.is_empty() {
        anyhow::bail!("{} tracts missing from assignments. First 5: {:?}",
            missing.len(), &missing[..missing.len().min(5)]);
    }
    Ok(())
}

/// Export in GerryChain v2.3 format.
/// Schema: {"assignment": {"GEOID": district_int, ...}}  (field name is singular)
/// Target: GerryChain >= 2.3. Earlier versions use different field names.
/// Compatible with GerryChain >= 2.3. Earlier versions use different field names;
/// see docs/specs/RPLAN.md for migration.
pub fn export_gerrychain_v23(plan: &RplanDocument) -> anyhow::Result<String> {
    // Map plan.assignments (plural) → JSON {"assignment": {...}} (singular)
    let assignment: Map<String, Value> = plan.assignments.iter()
        .map(|(k, v)| (k.clone(), json!(v)))
        .collect();
    let doc = json!({"assignment": assignment, "gerrychain_version_target": "2.3"});
    Ok(serde_json::to_string_pretty(&doc)?)
}

/// Import GerryChain v2.3 JSON back to RplanDocument assignments.
pub fn import_gerrychain_to_rplan(
    gc_json: &str,
    metadata: &crate::rplan::PlanManifest,
) -> anyhow::Result<RplanDocument> {
    let v: Value = serde_json::from_str(gc_json)?;
    let assignment = v["assignment"].as_object()
        .ok_or_else(|| anyhow::anyhow!("GerryChain JSON missing 'assignment' field"))?;
    let assignments: std::collections::HashMap<String, usize> = assignment.iter()
        .map(|(k, v)| {
            let dist = v.as_u64()
                .ok_or_else(|| anyhow::anyhow!("district value must be integer"))? as usize;
            Ok((k.clone(), dist))
        })
        .collect::<anyhow::Result<_>>()?;
    Ok(RplanDocument {
        rplan_version: "0.1".into(),
        metadata: metadata.clone(),
        assignments,
        geometry: None,
    })

/// Export tract assignments as GEOID,district CSV.
pub fn export_tracts_csv(plan: &RplanDocument) -> anyhow::Result<String> {
    let mut wtr = csv::Writer::from_writer(Vec::new());
    wtr.write_record(["geoid", "district"])?;
    let mut rows: Vec<_> = plan.assignments.iter().collect();
    rows.sort_by_key(|(k, _)| k.as_str());
    for (geoid, district) in rows {
        wtr.write_record([geoid.as_str(), &district.to_string()])?;
    }
    Ok(String::from_utf8(wtr.into_inner()?)?)
}
