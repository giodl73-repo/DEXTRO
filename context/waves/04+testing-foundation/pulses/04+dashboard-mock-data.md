## Current State

- Use generic element checks (`h1`, `button`, etc.)
- Don't test actual data rendering
- Don't verify maps load correctly
- Don't test tab content or filtering
- Can't run without real redistricting data

**Blockers**:
- Tests require real `outputs/us_2020_v1/` directory

- Can't test dashboard components in isolation

3. Verify all tabs, maps, tables, and charts render correctly
4. Test data filtering, sorting, and state switching
5. Validate CSV downloads and data accuracy

## Mock Run Generator Usage

### Command Line
```bash
# Generate mock run for testing
python tests/fixtures/generate_mock_run.py --states "vermont,alabama" --year 2020 --version test

# Generate for all states
python tests/fixtures/generate_mock_run.py --states "vermont,delaware,wyoming,alabama,california" --year 2020 --version test

# Custom output directory
python tests/fixtures/generate_mock_run.py --output-dir outputs/test_data --year 2020 --version test
```

### From Python/Pytest
```python
from tests.fixtures.generate_mock_run import generate_mock_run

@pytest.fixture(scope='session')
def mock_run():
    """Generate mock redistricting run for dashboard tests."""
    output_dir = Path('outputs/us_2020_test')

    # Generate mock data
    generate_mock_run(
        states=['vermont', 'alabama'],
        year='2020',
        version='test',
        output_dir=output_dir
    )

    yield output_dir

    # Cleanup after tests
    shutil.rmtree(output_dir)
```

## Example Dashboard Tests

### Test All Tabs Load
```python
def test_all_tabs_visible(page, mock_run):
    """Test all dashboard tabs are present and clickable."""
    page.goto(f"file:///{mock_run.parent.parent}/web/dashboard.html")

    tabs = ['Districts', 'Political', 'Demographics', 'Compactness', 'Urban', 'National']
    for tab in tabs:
        assert page.locator(f'text="{tab}"').is_visible()
        page.click(f'text="{tab}"')
        assert page.locator('.content.active').is_visible()
```

### Test District Table Data
```python
def test_district_table_data(page, mock_run):
    """Test district table displays correct data."""
    page.goto(f"file:///{mock_run.parent.parent}/web/dashboard.html")

    # Check table exists
    table = page.locator('table.district-table')
    assert table.is_visible()

    # Check header columns
    headers = table.locator('th').all_text_contents()
    assert 'District' in headers
    assert 'Population' in headers
    assert 'Compactness' in headers

    # Check data rows
    rows = table.locator('tbody tr')
    assert rows.count() > 0  # Has data

    # Check specific values (Vermont has 1 district)
    first_row = rows.nth(0)
    assert '1' in first_row.text_content()  # District 1
    assert '750' in first_row.text_content()  # Population ~750k
```

### Test Maps Load
```python
def test_maps_load(page, mock_run):
    """Test all maps load without 404 errors."""
    page.goto(f"file:///{mock_run.parent.parent}/web/dashboard.html")

    # State map
    state_map = page.locator('img[src*="all_districts.png"]').first
    assert state_map.is_visible()

    # Check image loaded (not broken)
    assert state_map.evaluate('img => img.naturalWidth > 0')
    assert state_map.evaluate('img => img.naturalHeight > 0')

    # Political tab (2020 only)
    page.click('text="Political"')
    political_map = page.locator('img[src*="partisan_lean.png"]')
    assert political_map.is_visible()
    assert political_map.evaluate('img => img.naturalWidth > 0')

    # Compactness tab
    page.click('text="Compactness"')
    pp_map = page.locator('img[src*="polsby_popper.png"]')
    assert pp_map.is_visible()
```

### Test State Switching
```python
def test_state_switching(page, mock_run):
    """Test state selector switches content correctly."""
    page.goto(f"file:///{mock_run.parent.parent}/web/dashboard.html")

    # Initial state (Vermont - 1 district)
    state_select = page.locator('select.state-select')
    assert state_select.input_value() == 'vermont'

    district_count = page.locator('.district-list-item').count()
    assert district_count == 1  # Vermont has 1 district

    # Switch to Alabama (7 districts)
    state_select.select_option('alabama')
    page.wait_for_timeout(500)  # Wait for content update

    district_count = page.locator('.district-list-item').count()
    assert district_count == 7  # Alabama has 7 districts
```
