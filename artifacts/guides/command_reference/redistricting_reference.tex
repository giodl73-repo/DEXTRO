\section*{Pipeline}

\textbf{Common:}
\begin{Verbatim}[fontsize=\small,frame=single]
run --version v1
  # All years (2-4h)
run --year 2020 --version v1
  # Single year (~1h)
run --version v1 --reset
  # Fresh run
run --version v1 --skip-states
  # National only (fast)
run --workers 12 --version v1
  # Custom workers
\end{Verbatim}

\textbf{Experimental:}
\begin{Verbatim}[fontsize=\small,frame=single]
run --partition-mode unweighted
  --version v1
run --run-type experiment
  --experiment-name test123
  --version v1
\end{Verbatim}

\section*{Pipeline Flags}

\textbf{Basic:}
\begin{itemize}\itemsep0pt
\item \texttt{--version v1} - Version ID
\item \texttt{--year \{2020|2010|2000|all\}}
\item \texttt{--workers N} - Parallel (def: 12)
\item \texttt{--dpi N} - Quality (def: 150)
\item \texttt{--reset} - Delete outputs
\item \texttt{--print-only} - Dry run
\end{itemize}

\textbf{Scope:}
\begin{itemize}\itemsep0pt
\item \texttt{--skip-states} - National only
\item \texttt{--states-only} - Skip national
\item \texttt{--skip-political} - No politics
\item \texttt{--skip-demographic} - No demos
\item \texttt{--reprocess} - Redo all states
\end{itemize}

\textbf{Experiments:}
\begin{itemize}\itemsep0pt
\item \texttt{--partition-mode \{edge-weighted|unweighted\}}
\item \texttt{--run-type \{production|experiment|test\}}
\item \texttt{--experiment-name NAME}
\item \texttt{--election-year \{2020|2016\}}
\item \texttt{--debug} - Debug mode
\end{itemize}

\textbf{Tests:}
\begin{itemize}\itemsep0pt
\item \texttt{-v} - Verbose
\item \texttt{-s} - Show prints
\item \texttt{-x} - Stop first fail
\item \texttt{--cov} - Coverage
\item \texttt{-k PATTERN} - Filter
\end{itemize}
