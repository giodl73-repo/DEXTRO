\section*{Pipeline}

\textbf{Common:}
\begin{Verbatim}[fontsize=\small,frame=single]
run -v v1
  # All years (2-4h)
run -y 2020 -v v1
  # Single year (~1h)
run -y 2020 -v v1 -st CA TX
  # Specific states
run -v v1 -r
  # Fresh run
run -v v1 -s nation
  # National only (fast)
run -w 12 -v v1
  # Custom workers
\end{Verbatim}

\textbf{Test/Debug:}
\begin{Verbatim}[fontsize=\small,frame=single]
runtest -y 2020 -v test
  # Test run (outputs/dev/)
runtest -y 2020 -v t -st VT
  # Single state test
run -p -v preview
  # Dry run
run -v v1 -d
  # Debug mode
\end{Verbatim}

\textbf{Experimental:}
\begin{Verbatim}[fontsize=\small,frame=single]
run -pm unweighted -v v1
run -rt experiment
  --experiment-name test123
  -v v1
\end{Verbatim}

\section*{Tests}

\section*{Pipeline Flags}

\textbf{Short Flags (all have long equivalents):}

\vspace{2pt}

\begin{small}
\begin{tabular}{|l|l|p{3cm}|l|p{3cm}|}
\hline
\textbf{Short} & \textbf{Long} & \textbf{Values} & \textbf{Default} & \textbf{Description} \\
\hline
\texttt{-h} & \texttt{--help} & - & - & Show all options \\
\texttt{-y} & \texttt{--year} & 2020/2010/2000/all & all & Census year(s) \\
\texttt{-v} & \texttt{--version} & string & v1 & Version ID \\
\texttt{-s} & \texttt{--stages} & data/states/nation & all & Pipeline stages \\
\texttt{-st} & \texttt{--states} & codes & all & States (CA TX NY) \\
\texttt{-w} & \texttt{--workers} & 1-24 & 12 & Parallel workers \\
\texttt{-r} & \texttt{--reset} & flag & - & Fresh run \\
\texttt{-p} & \texttt{--print-only} & flag & - & Dry run \\
\texttt{-d} & \texttt{--debug} & flag & - & Debug mode \\
\texttt{-ey} & \texttt{--election-year} & 2020/2016 & 2020 & Election data \\
\texttt{-pm} & \texttt{--partition-mode} & edge/unweighted & edge & METIS mode \\
\texttt{-rt} & \texttt{--run-type} & prod/test/exp & prod & Output type \\
\hline
\end{tabular}
\end{small}

\vspace{4pt}

\textbf{Long-only Flags:}
\begin{itemize}\itemsep0pt
\item \texttt{--dpi N} - Quality (def: 150)
\item \texttt{--minimum-boundary-length N} - Filter adjacencies (def: 10m)
\item \texttt{--skip-political} - No politics
\item \texttt{--skip-demographic} - No demos
\item \texttt{--reprocess} - Redo all states
\item \texttt{--experiment-name} - Experiment name
\end{itemize}

\textbf{Stage Examples:}
\begin{itemize}\itemsep0pt
\item \texttt{-s data} - Data only
\item \texttt{-s states nation} - Skip data
\item \texttt{-s nation} - Post-processing only
\end{itemize}

\textbf{Data Download Flags:}
\begin{itemize}\itemsep0pt
\item \texttt{-y, --year} - Census year (2000/2010/2020)
\item \texttt{-o, --output-dir} - Output directory
\end{itemize}

\textbf{Tests:}
\begin{itemize}\itemsep0pt
\item \texttt{-v} - Verbose
\item \texttt{-s} - Show prints
\item \texttt{-x} - Stop first fail
\item \texttt{--cov} - Coverage
\item \texttt{-k PATTERN} - Filter
\end{itemize}
