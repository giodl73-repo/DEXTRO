\section*{Git Basics}

\begin{tcolorbox}[colback=highlightbg,colframe=gitcolor,boxrule=1.5pt,arc=2pt,left=6pt,right=6pt,top=4pt,bottom=4pt]
\textbf{\large Essential Git Commands}

\vspace{4pt}

\begin{Verbatim}[fontsize=\large,frame=none]
gs          git status
ga .        git add .
gc "msg"    git commit -m
gp          git push
gl          git log (10)
gd          git diff
\end{Verbatim}
\end{tcolorbox}

\vspace{6pt}

\textbf{Additional Git:}
\begin{Verbatim}[fontsize=\normalsize,frame=single]
gco branch    git checkout
gb            git branch
\end{Verbatim}

\section*{Common Workflows}

\begin{tcolorbox}[colback=highlightbg,colframe=commandcolor,boxrule=1.5pt,arc=2pt,left=8pt,right=8pt,top=6pt,bottom=6pt,title={\large\textbf{1. Quick Commit \& Push}}]
\begin{Verbatim}[fontsize=\Large,frame=none]
gs
ga .
gc "your commit message"
gp
\end{Verbatim}
\end{tcolorbox}

\vspace{6pt}

\begin{tcolorbox}[colback=highlightbg,colframe=commandcolor,boxrule=1.5pt,arc=2pt,left=8pt,right=8pt,top=6pt,bottom=6pt,title={\large\textbf{2. Review Before Push}}]

\begin{tcolorbox}[colback=highlightbg,colframe=commandcolor,boxrule=1.5pt,arc=2pt,left=8pt,right=8pt,top=6pt,bottom=6pt,title={\large\textbf{3. Create Pull Request}}]

\end{Verbatim}
\end{tcolorbox}

\begin{tcolorbox}[colback=highlightbg,colframe=commandcolor,boxrule=1.5pt,arc=2pt,left=8pt,right=8pt,top=6pt,bottom=6pt,title={\large\textbf{4. Switch Branches}}]

\section*{Step-by-Step Guides}

\subsection*{First Time Setup}

\begin{Verbatim}[fontsize=\normalsize,frame=single]
# 1. Check git status
gs

# 2. Configure git (first time)
git config --global user.name
  "Your Name"
git config --global user.email
  "you@example.com"

# 3. Verify GitHub CLI
gh auth status
\end{Verbatim}

\subsection*{Making Changes}

\begin{Verbatim}[fontsize=\normalsize,frame=single]
# 1. Check current status
gs

# 2. See what changed
gd

# 3. Add all changes
ga .

# 4. Commit with message
gc "describe your changes"

# 5. Push to GitHub
gp
\end{Verbatim}

\subsection*{Working with PRs}

\begin{Verbatim}[fontsize=\normalsize,frame=single]
# List all PRs
ghprl

# View specific PR
ghprv 123

# Checkout PR locally
ghprc 123

# Create new PR
ghpr

# Merge PR
ghprm 123
\end{Verbatim}

\subsection*{View History}

\begin{Verbatim}[fontsize=\normalsize,frame=single]
# Last 10 commits
gl

# Full diff
gd

# Check status
gs
\end{Verbatim}

Write clear commit messages $\bullet$ Push often to backup your work
